1 /**
   2  * @class Core object class
   3  * @returns {Object}
   4  * @augments Class

   6 imageEditor.coreObject = Class.extend(
   7         /** @lends imageEditor.coreObject.prototype */
   8                 {
   9                     /**
  10                      * Constructor function for this class
  11                      * 
  12                      * @public
  13                      * @param {Object} params
  14                      * @constructs
  15                      */
  16                     __constructor: function(params) {
  17                         for (var t in params) {
  18                             this[t] = params[t];

  20                         this.eventListeners = {};
  21                     },
  22                     /**
  23                      * Setup an event
  24                      * 
  25                      * @public
  26                      * @param {String} event
  27                      * @param {Function} callback
  28                      */
  29                     on: function(event, callback) {
  30                         this.eventListeners[event] || (this.eventListeners[event] = []);
  31                         this.eventListeners[event].push(callback);
  32                     },
  33                     /**
  34                      * Trigger an event
  35                      * 
  36                      * @public
  37                      * @param {String} event
  38                      * @param {Object} callback
  39                      */
  40                     fire: function(event, callback) {
  41                         var r = this.eventListeners[event], u, i;
  42                         if (r) {
  43                             for (u = r.length, i = 0; i < u; i++) {
  44                                 r[i].apply(this, [callback || {}]);
  45                             }
  46                         }
  47                     }
  48                 }
  49         );
