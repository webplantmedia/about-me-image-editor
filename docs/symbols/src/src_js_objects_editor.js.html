 214                     /**
 215                      * Add an object into the editor's stage
 216                      * 
 217                      * @public
 218                      * @param {Object} obj
 219                      * @returns {Object}
 220                      */
 221                     addObject: function(obj) {
 222                         obj.layer = this.layer;
 223                         obj.stageWidth = this.width / this.scale;
 224                         obj.stageHeight = this.height / this.scale;
 225                         do {
 226                             obj.id = Math.floor(Math.random() * 1e4);
 227                         }
 228                         while (this.getObjectById(obj.id) !== null);
 229                         var t = null;
 230                         switch (obj.type) {
 231                             case "clipart":
 232                                 t = new imageEditor.uiClipart(obj);
 233                                 break;
 234                             case "image":
 235                                 t = new imageEditor.uiImage(obj);
 236                                 break;
 237                             case "text":
 238                                 t = new imageEditor.uiText(obj);
 239                                 break;
 240                             case "line":
 241                                 t = new imageEditor.uiLineShape(obj);
 242                                 break;
 243                             case "circle":
 244                                 t = new imageEditor.uiCircleShape(obj);
 245                                 break;
 246                             case "rectangle":
 247                                 t = new imageEditor.uiRectangleShape(obj);
 248                                 break;
 249                             case "triangle":
 250                                 t = new imageEditor.uiTriangleShape(obj);
 251                                 break;
 252                             case "background":
 253                                 t = new imageEditor.uiBackground(obj);
 254                         }
 255                         return this._addObject(t), t;
 256                     },
 257                     /**
 258                      * Return a pointer to the object specified by its id
 259                      * 
 260                      * @public
 261                      * @param {Integer} id
 262                      * @returns {Object}
 263                      */
 264                     getObjectById: function(id) {
 265                         for (var t = 0; t < this.items.length; ++t) {
 266                             if (id === this.items[t].id) {
 267                                 return this.items[t];
 268                             }
 269                         }
 270                         return null;
 271                     },
 272                     _addObject: function(obj) {
 273                         this.items[this.items.length] = obj;
 274                         obj.on("select", function() {
 275                             this.deselectAll(obj.id);
 276                             this.fire("item-selected", obj);
 277                         }.bind(this));
 278                         obj.on("progress-start", function(n) {
 279                             this.progressbar.show(n);
 280                         }.bind(this));
 281                         obj.on("progress-complete", function() {
 282                             this.progressbar.hide();
 283                         }.bind(this));
 284                         obj.on("ready", function() {
 285                             for (var n = 0; n < this.items.length; ++n) {
 286                                 if (!this.items[n].ready) {
 287                                     return false;
 288                                 }
 289                             }
 290                             this.fire("ready", this);
 291                         }.bind(this));
 292                         obj.on("error", function(n) {
 293                             $.ClassyNotty({
 294                                 title: 'Error',
 295                                 content: n,
 296                                 error: true,
 297                                 img: 'images/error.png'
 298                             });
 299                         }.bind(this));
 300                         obj.type !== "background" && this.updateBackgroundOrder();
 301                         obj.draw();
 302                         //obj.select();
 303                     },
